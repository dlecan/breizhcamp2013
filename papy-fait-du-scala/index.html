  val list = List("Cette", "conférence", "est", "toujours", "aussi", "géniale")

  def containsE(word: String): Boolean = word.contains("e")

  val listE = list filter containsE
           // list.filter(elt => containsE(elt))
           //> List(Cette, conférence, est, géniale)

  val getLength = (word: String) => word.length

  val res = listE map getLength //> List(5, 10, 3, 7)
